 C uses the stdlib header file for I/O functions as:
 #+begin_src c
 #inlcude <stdio> // contains standard function like printf() , scanf(), etc..

  #+begin_src c
 #inlcude <iostream> // contains standard function like cout , cin, etc..
 #+end_src
